\subsection{Quelque définition}
\begin{dfn}
Soit $(\Omega,\mathcal{F},\mathbb{P})$ un espace de probabilité et $\{\mathcal{F}_t,t\geq0\}$ une famille de sous-tribus de $\mathcal{F}$. On dit que $\{\mathcal{F}_t,t\geq0\}$ est une famille si c'est une famille croissante, au sens où $\mathcal{F}_s\subset\mathcal{F}_t$ si $s\leq t$.
\end{dfn}
\begin{rem}
	Si $\{\mathcal{F}_t,t\geq0\}$ est une filtration sur un espace de probabilité $(\Omega,\mathcal{F},\mathbb{P})$, on dit que $(\Omega,\{\mathcal{F},t\geq 0\},\mathbb{P})$ est un espace de probabilité filtré.
\end{rem}

\begin{dfn}
	Un processus stochastique $X(t,\omega)$ à valeurs dans un espace mesurable $(E,\mathcal{E})$ est une application de $T\times\Omega$ dans $E$ qui est mesurable par rapport à la mesure du produit $\lambda\cdot\mathbb{P}$ où $\lambda$ est la mesure de Lebesgue sur $T$. 
\end{dfn}
\begin{dfn}
	On dit que le processus $X$ est $(\mathcal{F}_t)-$adapté qi $X_t$ est $\mathcal{F}_t-$mesurable pour tout $t\geq0.$
\end{dfn}	
\begin{dfn}
	On dit que le processus $(X_t)$ est prévisible si, $X_t$ est $\mathcal{F}_{t^-}$-mesurable pour tout $t\geq0$, où
	\[
	\mathcal{F}_{t^-} =\sigma\Big(\bigcup_{s\leq t}\mathcal{F}_s\Big)
	\]
	est la tribu juste avant $t$ (limite à gauche de la filtration).
\end{dfn}
\begin{dfn}
	Une processus $X_t(\omega)$ sur $[0,T]\times\Omega$ est dite progressivement mesurable par rapport à la filtration $\mathcal{F}_t$ si pour tout $t\in[0,T]$ l'application:
	\[
	X: (s,\omega)\to X_t(\omega)
	\]
	de $[0,t]\times\Omega\to\mathbb{R}$ est $\mathcal{B}[0,t]\otimes\mathcal{F}_t-$mesurable.
\end{dfn}
\begin{thm}
Si $X_t$ un processus stochastique adaptée est continu à droite, il est progressivement mesurable.
\end{thm}
\begin{pr}
La démonstration complète du théorème est disponible dans la référence [8]
\end{pr}
\begin{dfn}
Un processus $X_t$ est dit continu (resp $\mathbb{P}$-presque sûrement continu) si pour tout $\omega\in\Omega$ (resp pour $\mathbb{P}$-presque tout $\omega\in\Omega$), la fonction 
\[
t \longmapsto X_t(\omega)
\] de $T=\mathbb{R}^+$ dans $E$ est continus.
\end{dfn}
\begin{prop}
	Si $(X_t)_{t\geq0}$ un processus stochastique défini sur un espace de probabilité $(\Omega,\mathcal{F},\mathbb{P})$ alors pour un $\omega\in\Omega$ fixé, l'application
	\[
	t\longmapsto X_t(\omega)
	\]
	est appelée la trajectoire (ou réalisation) du processus correspondant à l'issue $\omega$.
\end{prop}

Dans la suite, nous rappelons la définition de la martingale et du mouvement brownien.
\begin{dfn}
Une famille adaptée $(M_t)_{t\geq0}$ de variable aléatoires intégrables (c'est-à-dire vérifiant $\mathbb{E}(|M_t|)<\infty$ pour tout $t$) est:
\begin{itemize}
	\item une martingale si, pour tout $s\leq t$, $\mathbb{E}(M_t|\mathcal{F}_s)=M_s$.
	\item une sur-martingale si, pour tout $s\leq t$,$\mathbb{E}(M_t|\mathcal{F}_s)\leq M_s.$
	\item une sous-martingale si, pour tout $s\leq t$, $\mathbb{E}(M_t|\mathcal{F}_s)\geq M_s.$. 
\end{itemize}
\begin{rem}
	On déduit de cette définition que, si $(M_t)_{t\geq0}$ est une martingale, alors $\mathbb{E}(M_t)=\mathbb{E}(M_0)$, pour tout $t$.
\end{rem}
\end{dfn}

\begin{dfn}
	Un processus stochastique $W_t$ est un mouvement brownien ou un processus de Wiener standard si $W_0=0$ ( on dit que $W_t$ est issu de 0) e si pour tous réels $0<t_1<t_2<...<t_n$, les variables aléatoires $W_{t_1}-W_{t_0},...,B_{t_n}-W_{t_{n-1}}$ sont indépendantes et suivent un distribution gaussienne centré réduire (on dit que le brownien est standard si $m$=0 et $\sigma=1$) telle que 
	\[
	\begin{cases} 
		\mathbb{E}(W_{t+h} - W_t) = 0 \\ 
		\mathbb{E}(W_{t+h} - W_t)^2 = h 
	\end{cases}
	\]
\end{dfn}
\begin{rem}
Dans un cadre plus général, lorsque le mouvement brownien n'est pas "standard" (c'est-à-dire ni centré, ni réduit), on introduit deux paramètres fondamentaux : la dérive $(m)$ et la volatilité $(\sigma)$.

L'évolution de l'accroissement sur un intervalle $[t_{k-1},t_k]$ est alors régie par le système suivant:
\[
\begin{cases}
	\mathbb{E}(W_{t_k} - W_{t_{k-1}}) = m(t_k - t_{k-1}) \\
	\mathbb{E}\left((W_{t_k} - W_{t_{k-1}}) - m(t_k - t_{k-1})\right)^2 = \sigma^2(t_k - t_{k-1})
\end{cases}
\]
\end{rem}
\begin{prop}
	Tout mouvement brownien est un processus à accroissements indépendantes c'est-a-dire: Pour tous $s=t_0\leq t_1 \leq t_2 \leq ...\leq t_n$, les variables aléatoires $W_{t_h}-W_{t_{h-1}}(h=1,...,n)$ sont indépendantes et indépendantes de la tribu $\mathcal{F}_s$. 
\end{prop}
\begin{pr}
 La démonstration complète du théorème est disponible dans la référence [8]
\end{pr}
Donnons des exemples de martingales que l'on peut construire à partir du mouvement brownien.
\begin{prop}
	Si $(X_t)_{t\geq0}$ est un $\mathcal{F}_t$-mouvement brownien standard, alors
\begin{enumerate}
	\item $(X_t)_{t\geq0}$ est une $\mathcal{F}_t$-martingale.
	\item $(X_t^2-t)$ est une $\mathcal{F}_t$-martingale.
\end{enumerate}

\end{prop}
\subsection{Espace de Cameron-Martin et changement de mesure}
\begin{dfn}
	Soit $T>0$ fixé. On note $\Omega = C([0,T],\mathbb{R}^d)$ l'espace des fonctions continues $h : [0,T] \to \mathbb{R}^d$ telles que $h(0) = 0$.
	
	L'espace de Cameron–Martin $\mathbb{H}$ est l'ensemble des fonctions $h \in \Omega$ qui sont absolument continues et dont la dérivée (au sens de Lebesgue) est de carré intégrable :
	\[
	\mathbb{H} = \left\{ h \in C([0,T];\mathbb{R}^d) \;:\; h(0)=\mathbf{0},\; h \text{ absolument continue},\; \dot{h} \in L^2([0,T];\mathbb{R}^d) \right\}.
	\]
	On munit $\mathbb{H}$ du produit scalaire
	\[
	\langle h,k \rangle_{\mathbb{H}} := \int_{0}^{T} \dot{h}(t) \cdot \dot{k}(t) \, dt, \qquad h,k \in \mathbb{H},
	\]
	et de la norme associée $\|h\|_{\mathbb{H}} = \sqrt{\langle h,h \rangle_{\mathbb{H}}}$.
\end{dfn}

\begin{prop}
	L'espace $(\mathbb{H}, \langle\cdot,\cdot\rangle_{\mathbb{H}})$ est un espace de Hilbert séparable.
\end{prop}
\begin{dfn}
	On note $\mathcal{A}$ l'ensemble des processus $h:\Omega\times[0,T]\to\mathbb{R}^d,\{\mathcal{F}_t\}$-prévisible, tels que pour $\mathbb{P}$-presque tout $\omega$, la trajectoire $t\to h(\omega,t)$ appartienne à $\mathbb{H}$.
\end{dfn}
	\begin{dfn}Pour chaque $N>0$, on définit
	\[
	S_N:=\Big\{h\in\mathbb{H}:\int_{0}^{T}|\dot{h}(s)|^2ds\leq N \Big\}.
	\]
	L'ensemble $S_N$ est muni de la topologie faible induite par $\mathbb{H}$.
\end{dfn}
\begin{dfn}
Pour chaque $N>0$, on définit
\[
\mathcal{A}_{N}:=\big\{h\in\mathcal{A},h(\omega,\cdot)\in S_{N},\ \mathbb{P}\text{-p.s.}\big\}.
\]
Autrement dit, $\mathcal{A}_N$ est l'ensemble des processus prévisibles dont presque toutes les trajectoires sont dans $S_N$.
\end{dfn}
\begin{prop}[formule de Cameron-Martin]
	On note $W$ la mesure de Wiener définie sur l'espace des fonctions continues $C(\mathbb{R}_+,\mathbb{R})$, et $\mathbb{H}$ l'espace de Cameron-Martin associé. Soit $h\in\mathbb{H}$. Alors, pour toute fonction mesurable positive $\Phi$ sur $C(\mathbb{R}_+,\mathbb{R})$, on a
	\[
	\int \Phi(\omega+h)W(d\omega)=\int \Phi(\omega)\exp\big(\int_{0}^{+\infty}\dot{h}(s)d\omega (s)-\frac{1}{2}\int_{0}^{+\infty}|\dot{h}(s)|^2ds\big)W(d\omega).
	\]
\end{prop}

La formule de Cameron–Martin peut être vue comme un cas particulier du théorème de Girsanov, que nous présentons ci-dessous.

\begin{thm}[Théorème de Girsanov]
		Soit $(\Omega, \mathcal{F}, (\mathcal{F}_t)_{t \geq 0}, P)$ un espace probabilisé filtré satisfaisant les conditions usuelles. 
		Soit $Q$ une mesure de probabilité sur $(\Omega, \mathcal{F})$ équivalente à $P$ sur $\mathcal{F}_\infty$.
		
		On définit, pour tout $t \geq 0$, le processus de densité
		\[
		D_t = \frac{dQ}{dP}\Big|_{\mathcal{F}_t}.
		\]
		Alors $(D_t)_{t \geq 0}$ est une martingale positive, uniformément intégrable, que l'on suppose de plus continue.
		
		Dans ce cas, il existe une unique martingale locale continue $L$ telle que
		\[
		D_t = \mathcal{E}(L)_t, \quad t \geq 0,
		\]
		où $\mathcal{E}(L)$ désigne l'exponentielle stochastique associée à $L$.
		
		Soit enfin $M$ une martingale locale continue sous $P$. Alors, sous la mesure $Q$, le processus
		\[
		\widetilde{M}_t = M_t - \langle M, L \rangle_t, \quad t \geq 0,
		\]
		est une martingale locale continue.	
\end{thm}
Nous présentons à présent une conséquence importante du théorème de Girsanov, qui décrit explicitement l'effet d'un changement de probabilité sur un mouvement brownien.

\begin{cor}[Girsanov cas du mouvement brownien]
		Soit $(W_t)_{t \geq 0}$ un mouvement brownien standard sous $P$. 
		Soit $(\theta_t)_{t \geq 0}$ un processus adapté vérifiant les conditions d'intégrabilité appropriées (par exemple la condition de Novikov).
		
		On définit
		\[
		L_t = \int_0^t \theta_s \, dW_s,
		\quad \text{et} \quad
		D_t = \mathcal{E}(L)_t.
		\]
		
		On introduit une nouvelle mesure de probabilité $Q$ définie par
		\[
		\frac{dQ}{dP}\Big|_{\mathcal{F}_t} = D_t.
		\]
		
		Alors, sous la mesure $Q$, le processus
		\[
		\widetilde{W}_t = W_t - \int_0^t \theta_s \, ds
		\]
		est un mouvement brownien standard.

\end{cor}

\subsection{Dérivée au sens de Lion}

\begin{dfn}
	Soit $\mathcal{P}(\mathbb{R}^d)$ l'espace des mesures de probabilité sur $\mathbb{R}^d$.\\
	On note $\mathcal{P}_2(\mathbb{R}^d)$ est le sous-ensemble des mesure de probabilité sur $\mathbb{R}^d$ possédant un moment d'ordre deux fini, c'est-à-dire:
	\[
	\mathcal{P}_2(\mathbb{R}^d) = \left\{ \mu \in\mathcal{P}(\mathbb{R}^d): \mu(|\cdot|^2) = \int_{\mathbb{R}^d} |x|^2\mu(dx)<\infty
	\right\} 
	\]
	où $\mu(f):=\int fd\mu$ pour un fonction mesurable $f$
\end{dfn}
\begin{prop}
	$\mathcal{P}_2(\mathbb{R}^d)$ est un espace polonais muni de la distance de Wasserstein $\mathbb{W}_2$
\end{prop}
\begin{dfn}
		Soient $\mu$ et $\nu\in\mathcal{P}_2(\mathbb{R}^d)$, la distance de Wasserstein d'ordre 2 entre $\mu$ et $\nu$ est définie par la formule:
	\[
	\mathbb{W}_2(\mu, \nu) := \left( \inf_{\pi \in \mathcal{C}(\mu, \nu)} \int_{\mathbb{R}^d \times \mathbb{R}^d} |x - y|^2 \, \pi(dx, dy) \right)^{1/2}, 
	\quad \mu, \nu \in \mathcal{P}_2(\mathbb{R}^d),
	\]
	où $\mathcal{C}(\mu,\nu)$ désigne l'ensemble des couplages de $\mu$ et $\nu$; c'est-à-dire que $\pi\in\mathcal{C}(\mu,\nu)$ est une mesure de probabilité sur $\mathbb{R}^d\times\mathbb{R}^d$ telle que $\pi(\cdot\times\mathbb{R}^d) = \mu$ et $\pi(\mathbb{R}^d\times\cdot)=\nu$.
\end{dfn}
\begin{prop}
Soient $X$ et $Y$ deux variables aléatoires à valeurs dans $\mathbb{R}^d$, définies sur un même espace de probabilité, et telles que leurs lois $\mathcal{L}_X$ et $\mathcal{L}_Y \in\mathcal{P}_2(\mathbb{R}^d)$. Alors on a:
\[
\mathbb{W}_2(\mathcal{L}_X,\mathcal{L}_Y)\leq\big(\mathbb{E}|X-Y|^2\big)^{\frac{1}{2}}
\]
\end{prop}
\begin{dfn}
	Pour tout $\mu\in\mathcal{P}_2(\mathbb{R}^d)$, on définit l'espace tangent en $\mu$ par
	\[
	T_{\mu,2} = L^2(\mathbb{R})^d\longrightarrow \mathbb{R}^d :=\left\{\phi:\mathbb{R}^d\longrightarrow\mathbb{R}^d \text{ est mesurable avec }\int_{\mathbb{R}^d}|\phi(x)|^2\mu(dx)<\infty\right\}
	\]
	la norme associé est donnée par:
	\[
	\|\phi\|^2_{T_{\mu,2}}:=\int_{\mathbb{R}^d}|\phi(x)|^2\mu(dx)\] 
\end{dfn}
\begin{dfn}
	Soit $f:\mathcal{P}_2(\mathbb{R}^d)\longrightarrow\mathbb{R}^d$ un fonction continue, et $I_d$ la carte identité sur $\mathbb{R}^d$
	
	\begin{enumerate}
		\item Si, pour tout $\mu\in\mathcal{P}_2(\mathbb{R}^d)$, on a
		\[
		T_{\mu,2}\ni\phi \mapsto D^L_{\phi}f(\mu):=\lim_{\varepsilon\to 0}\frac{f(\mu\circ(I_d+\varepsilon\phi)^{-1})-f(\mu)}{\varepsilon}\in\mathbb{R}
		\]
		et que cette limite définit un fonctionnel linéaire borné, alors on dit que $f$ est intrinsèquement différentiable en $\mu$, et on note $D^L_{\phi}f(\mu)$ la dérivée intrinsèque de $f$ en $\mu$
		
		\item Si, pour tout $\mu\in\mathcal{R}_2(\mathbb{R})$, on a
		\[
		\lim_{\|\phi\|_{T_{\mu,2}} \to 0} \frac{|f(\mu \circ (I_d + \phi)^{-1}) - f(\mu) - D^L_\phi f(\mu)|}{\|\phi\|_{T_{\mu,2}}} = 0.
		\]
		alors on dit que $f$ est $L-$différentiable en $\mu$, et l'on note $D_\mu f(\mu)$ la $L-$dérivée (ou dérivée de Lions) de $f$ en $\mu$
		\\ Dans ce cas, par le théorème de représentation de Riez, l'élément unique $D^Lf(\mu)\in T_{\mu,2}$ satisfaisant
		\[
		\langle D^L f(\mu), \phi \rangle_{T_{\mu,2}} := \int_{\mathbb{R}^d} \langle D^L f(\mu)(x), \phi(x) \rangle \mu(dx) = D^L_\phi f(\mu), \quad \forall \phi \in T_{\mu,2},
		\]
		\item  On écrit \( f \in C^1(\mathcal{P}_2(\mathbb{R}^d)) \) si \( f \) est \( L \)-différentiable en tout point \( \mu \in \mathcal{P}_2(\mathbb{R}^d) \), et la dérivée \( L \)-dérivée admet version \( D^L f(\mu)(x) \) jointe continu en \( (x, \mu) \in \mathbb{R}^d \times \mathcal{P}_2(\mathbb{R}^d) \). Si de plus, \( D^L f(\mu)(x) \) est bornée, on dénote \( f \in C_b^1(\mathcal{P}_2(\mathbb{R}^d)) \).
		
		Pour une fonction à valeurs vectorielles $f = (f_i)$, ou une fonction à valeurs matricielles $f = (f_{ij})$ avec des composantes $L$-différentiables, on écrit
		\[
		D^L_\phi f(\mu) = (D^L_\phi f_i(\mu)), \quad \text{ou} \quad D^L_\phi f(\mu) = (D^L_\phi f_{ij}(\mu)), \quad \mu \in \mathcal{P}_2(\mathbb{R}^d).
		\] 
	\end{enumerate}
\end{dfn}
\subsection{Quelques inégalités}
\begin{lem}[Inégalité de Gronwall]
	Soit $g$ un fonction continue telle que pour tout $t\geq0$, on a:
	pour tout $t\in[0,T]$
	\[
	 0\leq g(t)\leq \alpha(t)+\beta\int_{0}^{t}g(t)ds, \text{ avec } \alpha\geq0,\beta\geq0
.	\]
Pour une constante $\beta\geq0$ et pour une fonction $\alpha:[0,T]\to\mathbb{R}$ intégrable par rapport à la mesure de Lebesgue. On a alors
pour tout $t\in[0,T]$
\[
 0\leq g(t)\leq \alpha(t) +\beta\int_{0}^{t}\alpha(s)\exp(\beta(t-s))ds.
\]
En particulier, si $\alpha$ est une fonction constante, on trouve pour tout $t\in[0,T]$
\[
g(t)\leq\alpha\exp(\beta).
\]
\end{lem}
\begin{lem}[Inégalité de Hölder]
		Soient $p,q > 1$ tels que $\frac{1}{p} + \frac{1}{q} = 1$. 
	Soient $X$ et $Y$ deux variables aléatoires (ou fonctions mesurables) telles que 
	$|X|^p$ et $|Y|^q$ soient intégrables.
	
	Alors, on a l’inégalité suivante :
	\[
	\mathbb{E}(|XY|) \leq \left(\mathbb{E}(|X|^p)\right)^{\frac{1}{p}} 
	\left(\mathbb{E}(|Y|^q)\right)^{\frac{1}{q}}.
	\]
	
\end{lem}
\begin{lem}[Inégalité de Cauchy-Schwarz]
		Soient $X$ et $Y$ deux variables aléatoires de carré intégrable.
	
	Alors, on a :
	\[
	\mathbb{E}(|XY|) \leq \left(\mathbb{E}(X^2)\right)^{\frac{1}{2}} 
	\left(\mathbb{E}(Y^2)\right)^{\frac{1}{2}}.
	\]
\end{lem}

\begin{thm}[Burkholder--Davis--Gundy]
	Soit $(\Omega,\mathcal{F},(\mathcal{F}_t)_{t\geq 0},\mathbb{P})$ un espace probabilisé filtré satisfaisant les conditions usuelles. On note $\langle M,M\rangle$ la variation quadratique d’une martingale locale continue $M$.
	
	Alors, pour tout $p \in ]0,+\infty[$, il existe deux constantes positives $c_p$ et $C_p$, ne dépendant que de $p$, telles que, pour toute martingale locale continue $M=(M_t)_{t\geq 0}$ vérifiant $M_0 = 0$, on ait :
	\[
	c_p \, \mathbb{E}\big(\langle M,M\rangle_\infty^{\frac{p}{2}}\big)
	\;\leq\;
	\mathbb{E}\big((M^*_\infty)^p\big)
	\;\leq\;
	C_p \, \mathbb{E}\big(\langle M,M\rangle_\infty^{\frac{p}{2}}\big).
	\]
	
	Ici, $M^*_\infty$ désigne le supremum maximal de la martingale, défini par
	\[
	M^*_\infty = \sup_{t \geq 0} |M_t|,
	\]
	et $\langle M,M\rangle_\infty$ représente la limite de la variation quadratique lorsque $t$ tend vers l’infini, c’est-à-dire
	\[
	\langle M,M\rangle_\infty = \lim_{t \to \infty} \langle M,M\rangle_t.
	\]
	
	Les constantes $c_p$ et $C_p$ sont dites universelles, dans le sens où elles ne dépendent que du paramètre $p$ et peuvent être choisies de manière uniforme pour toutes les martingales locales continues définies sur un espace probabilisé filtré quelconque satisfaisant les conditions usuelles.
\end{thm}
\begin{prop}
	Soit $p \geq 2$. Il existe une constante $C_p$ telle que, pour toute martingale locale continue $M$ vérifiant $M_0 = 0$, on ait :
	\[
	\mathbb{E}\big[(M^*_\infty)^p\big] \leq C_p \, \mathbb{E}\big(\langle M,M\rangle_\infty^{p/2}\big).
	\]
	
\end{prop}

\begin{prop}
	Soit $p \geq 4$. Il existe une constante $c_p$ telle que :
	\[
	c_p \, \mathbb{E}\big(\langle M,M\rangle_\infty^{p/2}\big) \leq \mathbb{E}\big[(M^*_\infty)^p\big].
	\]
\end{prop}

\section{Les équations différentielles stochastiques}
Dans cette section, nous rappelons les équations différentielles stochastiques classiques ainsi que les équations différentielles stochastiques de type McKean–Vlasov, en précisant les principales hypothèses nécessaires à leur étude.

Dans toute la suite, nous nous plaçons sur un espace de probabilité filtré $(\Omega,\mathcal{F},(\mathcal{F})_{t\geq0}, \mathbb{P})$ satisfaisant les conditions usuelles.
\subsection{Formule d'Itô}
\begin{dfn}
	On appelle processus d'Itô, un processus $(X_t)_{0\leq t\leq T}$ à valeurs dans $\mathbb{R}$ tel que
	\[
	\mathbb{P}p.s \forall t\leq T\text{   } X_t = X_0 + \int_{0}^{t}K_sds+\int_{0}^{t}H_sdW_s
	\]
	avec:
	\begin{itemize}
		\item $X_0$ $\mathcal{F}_0$-mesurable.
		\item $(K_t)_{0\leq t\leq T}$ et $(H_s)_{0\leq t\leq T}$ des processus adaptés à $\mathcal{F}_t$.
		\item $\int_{0}^{T}|K_s|ds<+\infty \text{  }\mathbb{P}p.s$.
		\item $\int_{0}^{T}|H_s|^2ds<+\infty\text{  }\mathbb{P}p.s$.
	\end{itemize}
\end{dfn}

La proposition suivante montre qu'une martingale continue de variation finie est nécessairement constante.
\begin{prop}
Soit \((M_t)_{0 \leq t \leq T}\) une martingale continue telle que 
\[
M_t = \int_0^t K_s \, ds, \quad \text{avec } \mathbb{P}\text{ p.s. } \int_0^T |K_s| \, ds < +\infty,
\]
alors :
\[
\mathbb{P}\text{ p.s. } \quad \forall t \leq T, \, M_t = 0.
\]
\end{prop}

Le théorème suivant, appelé formule d’Itô, est l’outil central du calcul stochastique :
\begin{thm}[ Formule Itô]
		Soit $(X_t)_{0 \le t \le T}$ un processus d'Itô de la forme
		\[
		X_t = X_0 + \int_0^t K_s \, ds + \int_0^t H_s \, dW_s,
		\]
		et $f$ une fonction deux fois continûment différentiable sur $\mathbb{R}$. On a
		\[
		f(X_t) = f(X_0) + \int_0^t f'(X_s) \, dX_s + \frac{1}{2} \int_0^t f''(X_s) \, d\langle X, X \rangle_s,
		\]
		où, par définition,
		\[
		\langle X, X \rangle_t = \int_0^t H_s^2 \, ds,
		\]
		et
		\[
		\int_0^t f'(X_s) \, dX_s = \int_0^t f'(X_s) K_s \, ds + \int_0^t f'(X_s) H_s \, dW_s.
		\]
		De même, si $f(t,x) \mapsto f(t,x)$ est une fonction deux fois différentiable en $x$ et une fois différentiable en $t$, ces dérivées étant continues en $(t,x)$ (on dit dans ce cas que $f$ est de classe $C^{1,2}$), on a
		\[
		f(t, X_t) = f(0, X_0) + \int_0^t f_s'(s, X_s) \, ds 
		+ \int_0^t f_x'(s, X_s) \, dX_s + \frac{1}{2} \int_0^t f_{xx}''(s, X_s) \, d\langle X, X \rangle_s.
		\]
\end{thm}
\subsection{Équation différentielle stochastique}
\begin{dfn}
Soient $d$ et $m$ des entiers positifs, et soient $\sigma$ et $b$ des fonctions mesurables localement bornées définies sur $\mathbb{R}_+\times\mathbb{R}^d$ et prenant leurs valeurs respectivement dans $M_{d\times m}(\mathbb{R})$ et dans $\mathbb{R}^d$.

L'équation différentielle stochastique associée à $(b,\sigma)$ et s'écrit formellement
\begin{equation}\label{eq:1.2.1}
X_t = Z+\int_{0}^{t}b(s,X_s)ds +\int_{0}^{t}\sigma(s,X_s)dW_s.
\end{equation}
Une solution de (\ref{eq:1.2.1}) porte le nom diffusion.
\end{dfn}
 Nous allons étudier quelques propriétés des solutions de ces équations.
 \begin{dfn}
 On considère deux fonction $b$ et $\sigma$, $b:\mathbb{R}^+\times\mathbb{R}\textrightarrow \mathbb{R},\sigma:\mathbb{R}^+\times\mathbb{R}\textrightarrow \mathbb{R}$, $X_0$ un variable aléatoire $\mathcal{F}_0$-mesurable et $(W_t)_{t\geq0}$ un $\mathcal{F}_t$-mouvement brownien. Une solution à l'équation (\ref{eq:1.2.1}) est un processus stochastique (diffusion) $(X_t)_{t\geq0}$ continu $\mathbb{F}_t$-adapté, qui vérifie:
 \begin{itemize}
 	\item Pour tout $t\geq0$, les intégrales $\int_{0}^{t}b(s,X_s)ds$ et $\int_{0}^{t}\sigma(s,X_s)dW_s$ ont un sens
 	\[
 	\int_{0}^{t}|b(s,X_s)|ds<+\infty\text{ et }\int_{0}^{t}|\sigma(s,X_s)|^2dW_s<+\infty \mathbb{P}\text{  }p.s
 	\]
 	\item $(X_t)_{t\geq0}$ vérifie (\ref{eq:1.2.1}) c'est-à-dire:
 	\[
 	\forall t\geq0\text{    }\mathbb{P}\text{    }p.s. X_t=Z+\int_{0}^{t}b(s,X_s)ds +\int_{0}^{t}\sigma(s,X_s)dW_s.
 	\]
 \end{itemize}
 \end{dfn}
 \begin{rem}
 	On ré-écrit formellement l'équation (3.10) sous la forme:
 	\[
 	\begin{cases}
 		dX_t &= b(t, X_t)dt + \sigma(t, X_t)dW_t \\
 		X_0 &= Z
 	\end{cases}
 	\]
 \end{rem}
Le théorème suivant donne des conditions suffisantes sur \( b \) et \( \sigma \) pour avoir un résultat d’existence et d’unicité pour (\ref{eq:1.2.1}).

\begin{thm}
	Si \( b \) et \( \sigma \) sont des fonctions continues, telles qu'il existe \( K < +\infty \), avec
	\begin{enumerate}
		\item \( |b(t,x) - b(t,y)| + |\sigma(t,x) - \sigma(t,y)| \leq K|x - y| \)
		\item \( |b(t,x)| + |\sigma(t,x)| \leq K(1 + |x|) \)
		\item \( \mathbb{E}(Z^2) < +\infty \),
	\end{enumerate}
	alors, pour tout \( T \geq 0 \), l'équation (\ref{eq:1.2.1}) admet une solution unique dans l'intervalle \([0,T]\). De plus cette solution \((X_s)_{0 \leq s \leq T}\) vérifie
	\[
	\mathbb{E} \left( \sup_{0 \leq s \leq T} |X_s|^2 \right) < +\infty.
	\]
	L'unicité signifie que si \((X_t)_{0 \leq t \leq T}\) et \((Y_t)_{0 \leq t \leq T}\) sont deux solutions de l'équation (\ref{eq:1.2.1}), alors \( \mathbb{P}.p.s. \ \forall 0 \leq t \leq T, \ X_t = Y_t. \)
\end{thm}

Le théorème suivant établit un lien entre les différentes notions d'existence et d'unicité.

\begin{thm}[Yamada–Watanabe]
	Si l’existence faible et l’unicité trajectorielle ont lieu, alors l’unicité faible a également lieu. De plus, pour tout choix de l’espace de probabilité filtré \((\Omega, \mathcal{F}, (\mathcal{F}_t), \mathbb{P})\) et du mouvement brownien \((\mathcal{F}_t)\)-adapté $W$, il existe pour tout \(x \in \mathbb{R}^d\) une solution forte (unique) de l’équation (\ref{eq:1.2.1}).
\end{thm}

\subsection{Équation différentielle stochastique de type McKean-Vlasov}
Nous rappelons l’équation différentielle stochastique de McKean–Vlasov, définie sur l’espace euclidien $(\mathbb{R}^d, \langle \cdot, \cdot \rangle ,|\cdot|)$. 
Soit $(\Omega,\mathcal{F}_t,\{\mathcal{F}_t\}_{t\geqslant 0},\mathbb{P})$ un espace de probabilité filtré complet et $\mathcal{L}_{X^\varepsilon_t}$ désigne la loi du processus stochastique $X^\varepsilon_t$.
\begin{dfn}
	On appelle équation différentielle stochastique de type Mckean-Vlasov (MV-SDE) sur $(\mathbb{R}^d, \langle \cdot, \cdot \rangle ,|\cdot|)$ une équation de la forme:
	\begin{equation}\label{eq:1.2.2}
		dX_t^\varepsilon = b_t(X_t^\varepsilon,\mathcal{L}(X_t^\varepsilon))dt +\sqrt{\varepsilon}\sigma_t(X_t^\varepsilon,\mathcal{L}(X_t^\varepsilon))dW_t,\text{    } X_0^\varepsilon=x
	\end{equation}
	où $\varepsilon > 0$ est appelé le para
	mètre d'échelle.Ici, $W_t$ est une mouvement brownien dimension $d$ défini sur un espace de probabilité filtré et :
	\[
	\begin{aligned}
	&b:[0,\infty) \times \mathbb{R}^d\times\mathcal{P}_2(\mathbb{R}^d)\longrightarrow\mathbb{R}^d\\
	&\sigma:[0,\infty) \times \mathbb{R}^d\times\mathcal{P}_2(\mathbb{R}^d)\longrightarrow\mathbb{R}^{d\otimes d}
	\end{aligned}
	\]
\end{dfn}
Nous considérons les hypothèses suivantes.
\subsubsection{Hypothèses sur les coefficients}
\textbf{(H1)} Les coefficients $b : [0, \infty) \times \mathbb{R}^d \times \mathcal{P}_2(\mathbb{R}^d) \to \mathbb{R}^d$ et $\sigma : [0, \infty) \times \mathbb{R}^d \times \mathcal{P}_2(\mathbb{R}^d) \to \mathbb{R}^{d \otimes d}$ sont continus. Il existe une fonction croissante $K : [0, \infty) \to [0, \infty)$ telle que
\begin{equation}\label{eq:1.2.3}
	\begin{aligned}
	&\max\left\{ \|\nabla b_t(\cdot, \mu)(x)\|, \| D^L b_t(x, \cdot)(\mu) \|_{T_{\mu,2}} \right\} \leq K(t),\\ &\quad t \geq 0,\ x \in \mathbb{R}^d,\ \mu \in \mathcal{P}_2(\mathbb{R}^d),
	\end{aligned}
\end{equation}
\begin{equation}\label{eq:1.2.4}
	\begin{aligned}
	&\|\sigma_t(x, \mu) - \sigma_t(y, \nu)\|\leq K(t)(|x - y| + \mathbb{W}_2(\mu, \nu)),\\ &\quad t \geq 0,\ x, y \in \mathbb{R}^d,\ \mu, \nu \in \mathcal{P}_2(\mathbb{R}^d),
	\end{aligned}
\end{equation}
et
\begin{equation}\label{eq:1.2.5}
	|b_t(0, \delta_0)| + \|\sigma_t(0, \delta_0)\| \leq K(t), \quad t \geq 0.
\end{equation}

\textbf{(H2)} Le coefficient $b_t(x, \mu)$ est différentiable par rapport à $x$ et $\mu$, respectivement, et les fonctions dérivées satisfont
\begin{equation}\label{eq:1.2.6}
	\|\nabla b_t(., \mu)(x) - \nabla b_t(., \nu)(y)\| \leq K(t)(|x - y| + \mathbb{W}_2(\mu, \nu)),
\end{equation}
\[
\begin{aligned}
	&\left| \mathbb{E}\left[ \langle D^L b_t(x, \cdot)(\mathcal{L}_X)(X), \phi \rangle \right] 
	- \mathbb{E}\left[ \langle D^L b_t(y, \cdot)(\mathcal{L}_Y)(Y), \phi \rangle \right] \right| \\
	&\leq K(t)\left( |x - y| + \mathbb{W}_2(\mathcal{L}_X, \mathcal{L}_Y) + \left( \mathbb{E}[|X - Y|^2] \right)^{1/2} \right) \left( \mathbb{E}[|\phi|^2] \right)^{1/2}
\end{aligned}
\]

pour tout $t \geq 0$, $x, y \in \mathbb{R}^d$, $\mu, \nu \in \mathcal{P}_2(\mathbb{R}^d)$ et $X, Y, \phi \in L^2(\Omega \to \mathbb{R}^d, \mathbb{P}$
\begin{rem}
	D'après (H1), on a pour tout $t\geq0$ $x,y\in\mathbb{R}^d,\mu,\nu\in\mathcal{P}_2(\mathbb{R}^d)$ que:
	\begin{equation}\label{eq:1.2.7}
		|b_t(x, \mu) - b_t(y, \nu)| \leq K(t)(|x - y| + \mathbb{W}_2(\mu, \nu)).
	\end{equation}
\end{rem}

Intuitivement, lorsque le paramétré $\varepsilon$ tend vers 0 dans (1.2.6), le terme de diffusion disparaître et on obtient l'équation différentielle ordinaire suivante:
\begin{equation}\label{eq:1.2.8}
	dX_t^0=b_t(X^0_t,\delta_{X_t^0})dt.
\end{equation}
avec la même donnée initiale que (\ref{eq:1.2.2}), c'est-à-dire $X_0^0=x$. Comme $x$ est déterministe, on déduit que $\delta_{X^0_\cdot}$ est une mesure de Dirac centrée sur la trajectoire $X^0$.

\section{Principe de grandes déviations}
Le principe de grandes déviations (PGD) décrit le comportement asymptotique, lorsque $\varepsilon \to 0$, d’une famille de mesures de probabilité $(\mu_{\varepsilon})$ définies sur un espace mesurable $(E,\mathcal{B})$, à l’aide d’une fonction de taux. Plus précisément, cette caractérisation repose sur des bornes exponentielles asymptotiques supérieures et inférieures des probabilités $\mu_{\varepsilon}(\Gamma)$, pour des ensembles mesurables $\Gamma \subset E$.
Dans toute la suite, l’espace $E$ est supposé être un espace topologique, de sorte que les notions d’ouverts et de fermés y sont bien définies. Nous nous intéressons principalement aux ensembles boréliens, c’est-à-dire aux éléments de la tribu borélienne $\mathcal{B}(E)$.
Afin d’éviter toute difficulté liée à la mesurabilité, tous les espaces probabilisés considérés dans ce travail sont supposés complétés. Par abus de notation, $\mathcal{B}(E)$ désignera toujours la tribu borélienne complétée.
\begin{dfn}
Une fonction $I$ est une application semi-continue inférieurement
\[
I:E\longrightarrow[0,+\infty)
\] (c'est-à-dire que, pour tout $\alpha\in[0,+\infty)$, l'ensemble de niveau
\[
\Phi_I(\alpha) = \{ x\in E:I(x)\leq\alpha\}
\] est un sous-ensemble fermé de $E$).

Une bonne fonction de taux est une fonction de taux telle que tous les ensembles de niveau $\Phi_I(\alpha)$ soient des sous-ensembles compacts de $E$.
\end{dfn}
\begin{rem}
Si $E$ est un espace métrique, la propriété de semi-continuité inférieure peut être vérifiée à l'aide des suites, c'est-à-dire que $I$ est semi-continue inférieurement si et seulement si
\[
\liminf_{x_n \to x} I(x_n) \geq I(x),\text{      pour tout }x\in E.
\]
\end{rem}

La définition suivante donne le principe de grandes déviations dans un cadre général.
\begin{dfn}[Principe de grandes déviations]
	Soit $E$ un espace métrique muni de sa tribu borélienne $\mathcal{B}(E)$, et soit $(\mu_\varepsilon)_{\varepsilon>0}$ une famille de mesures de probabilité sur $E$.
	
	On dit que $(\mu_\varepsilon)$ satisfait un principe de grandes déviations avec une bonne fonction de taux $I$ si, pour tout $\Gamma \in \mathcal{B}(E)$,
	\[
	-\inf_{x \in \Gamma^\circ} I(x) 
	\le \liminf_{\varepsilon \to 0} \varepsilon \log \mu_\varepsilon(\Gamma) 
	\le \limsup_{\varepsilon \to 0} \varepsilon \log \mu_\varepsilon(\Gamma) 
	\le -\inf_{x \in \overline{\Gamma}} I(x).
	\]
	Ici, $\Gamma^\circ$ désigne l'intérieur de $\Gamma$ et $\overline{\Gamma}$ son adhérence.
\end{dfn}
Les inégalités ci-dessus sont appelées respectivement borne inférieure et borne supérieure du principe de grandes déviations.

De manière heuristique, cette définition signifie que la probabilité des événements rares décroît exponentiellement avec une vitesse gouvernée par la fonction de taux $I$.

Le résultat suivant permet de transférer un principe de grandes déviations à travers une application continue. La fonction de taux associée est alors obtenue en minimisant la fonction de taux initiale sur l’ensemble des antécédents de chaque point.

\begin{thm}
	Soient $E$ et $F$ deux espaces topologiques de Hausdorff et 
	\[
	f: E\longrightarrow F
	\] une application continue.
	Soit $I:E\longrightarrow [0,\infty]$ une bonne fonction de taux.
	\begin{itemize}
		\item[(a)] Pour tout $y\in F$, on définit:
		\[
		I'(y)=\inf\{I(x):x\in E,y=f(x)\}.
		\]
		Alors $I'$ est une bonne fonction de taux sur $F$, où, comme d'habitude, l'infimum sur l'ensemble vide pris égal à +$\infty$.
		\item[(b)] Si $I$ gouverne le principe de grandes déviations associé à une famille de mesures de probabilité $(\mu_\varepsilon)$ sur $E$, alors $I'$ gouverne le principe de grandes déviations associé à la famille de mesures $(\mu_\varepsilon\circ f^{-1})$ sur $F$.
	\end{itemize}
\end{thm}
Le théorème suivant, issu du livre [de Dembo et Zeitouni], établit un lien fondamental entre le mouvement brownien et le principe de grandes déviations. Il constitue un exemple classique de processus stochastique satisfaisant un principe de grandes déviations.

Soit $(W_t)_{t\in[0,1]}$ un mouvement brownien standard à valeurs dans $\mathbb{R}^d$. On considère le processus
\[
W^\varepsilon(t)=\sqrt{\varepsilon}\, W_t,
\]
et on note $\nu^\varepsilon$ la mesure de probabilité induite par $W^\varepsilon(\cdot)$ sur l’espace $C_0([0,1],\mathbb{R}^d)$ des fonctions continues nulles en $0$.

\begin{thm}[Théorème de Schilder]
	La famille $(\nu^\varepsilon)_{\varepsilon>0}$ satisfait un principe de grandes déviations dans $C_0([0,1],\mathbb{R}^d)$ avec une bonne fonction de taux $I_w$ définie par
	\[
	I_W(h) =
	\begin{cases}
		\displaystyle \frac{1}{2} \int_{0}^{1} |\dot{h}(t)|^2 dt & \text{si } h \in \mathbb{H}, \\
		+\infty & \text{sinon},
	\end{cases}
	\]
	où $\mathbb{H}$ désigne l’espace de Cameron-Martin.
\end{thm}
\paragraph{Principe de contraction}
Le théorème suivant est connu sous le nom de principe de contraction. Il permet de déduire un principe de grandes déviations pour l’image d’une famille de mesures par une application continue, à partir du principe de grandes déviations vérifié sur l’espace de départ. La nouvelle fonction de taux s’obtient alors en prenant, pour chaque point d’arrivée, l’infimum de l’ancienne fonction de taux sur l’ensemble de ses antécédents.

\begin{thm}\label{thm:contraction}
	Soient $E$ et $F$ deux espaces topologiques de Hausdorff, et soit
	$f:E\longrightarrow F$ une application continue.
	Soit $I:E\longrightarrow [0,+\infty]$ une bonne fonction de taux.
	\begin{itemize}
		\item[(a)] Pour tout $y\in F$, on définit
		\[
		I'(y)=\inf\{I(x): x\in E,\ f(x)=y\},
		\]
		en convenant que l'infimum d'un ensemble vide vaut $+\infty$.
		Alors $I'$ est une bonne fonction de taux sur $F$.
		
		\item[(b)] Si $I$ gouverne le principe de grandes déviations associé à une famille de mesures de probabilité $(\mu_\varepsilon)$ sur $E$, alors $I'$ gouverne le principe de grandes déviations associé à la famille $(\mu_\varepsilon\circ f^{-1})$ sur $F$.
	\end{itemize}
\end{thm}
Ce résultat montre que le principe de grandes déviations est stable par application continue. La nouvelle fonction de taux est obtenue en minimisant la fonction de taux initiale sur l’ensemble des antécédents de chaque point.
\begin{rem}
 	\begin{enumerate}
 		\item Ce théorème reste valable si $\mathcal{B}_E\nsubseteq \mathcal{B}$ (c'est-à-dire quand $\mathcal{B}_E$ n'est pas inclus dans $\mathcal{B}$), puisque pour tout ensemble mesurable $A\subset F$, on a toujours
 		\[
 		f^{-1}(A^\circ) \subset (f^{-1}(A))^\circ \text{ et } f^{-1}(\overline{A}) \subset \overline{f^{-1}(A)}
 		\]
 		\item Les bonnes de la partie (b) du théorème \ref{thm:contraction} tiennent même si $I$ n'est pas une bonne fonction de taux. Notez cependant que sans cette propriété, $I$ peut cesser d'être une fonction de taux (cf. l'exemple $E=F=\mathbb{R},\text{ } I(x)=0\text{ et } f(x)=e^x$).
 		\item Le théorème est vérifier dès lorsque $f$ est continue sur $D_I$. Cette observation permet d'envisager une extension du principe de contraction à une classe d'application approximativement continues
 	\end{enumerate}
\end{rem}

\paragraph{Équivalence exponentielle} 
Afin d'étendre le principe de contraction au-delà du cas continu, il semble naturel d'envisager des approximations par des fonctions continues. Il est toutefois utile d'aborder une question plus vaste : à quelle condition le PGD (Principe de Grandes Déviations) pour une famille de lois $\{\tilde{\mu}_\epsilon\}$ peut-il être déduit du PGD d'une famille $\{\mu_\epsilon\}$ ? L'application aux contractions approchées découlera de ces résultats généraux.
\begin{dfn}\label{def_exp_equ}
	Soit $(\mathcal{Y},d)$ un espace métrique. Les mesures de probabilité $\{\mu_\epsilon\}$ et $\{\tilde{\mu}_\epsilon\}$ sur $\mathcal{Y}$ sont dites exponentiellement équivalentes s'il existe des espaces de probabilité $\{(\Omega, \mathcal{B}_\epsilon, P_\epsilon)\}$ et deux familles de variables aléatoires $\{Z_\epsilon\}$ et $\{\tilde{Z}_\epsilon\}$ à valeurs dans $\mathcal{Y}$, de lois jointes $\{P_\epsilon\}$ et de lois marginales respectives $\{\mu_\epsilon\}$ et $\{\tilde{\mu}_\epsilon\}$, telles que la condition suivante soit satisfaite :
	Pour tout $\delta> 0$, l'ensemble $\{\omega : (\tilde{Z}_\epsilon, Z_\epsilon) \in \Gamma_\delta\}$ est $\mathcal{B}_\epsilon$-mesurable, et 
	\begin{equation}
		\limsup_{\epsilon \to 0} \epsilon \log P_\epsilon(\Gamma_\delta) = -\infty,
	\end{equation}
	où
	\begin{equation}
		\Gamma_\delta := \{(\tilde{y}, y) : d(\tilde{y}, y) > \delta\} \subset \mathcal{Y} \times \mathcal{Y}.
	\end{equation}
\end{dfn}
\begin{rem}
	\begin{itemize}
		\item [(a)] Les variables aléatoires $\{Z_\epsilon\}$ et $\{\tilde{Z}_\epsilon\}$ de la Définition \ref{def_exp_equ} sont dites exponentiellement équivalentes.
		\item [(b)] Il est relativement aisé de vérifier que la condition de mesurabilité est satisfaite dès lors que $\mathcal{Y}$ est un espace séparable, ou lorsque les lois $\{P_\epsilon\}$ sont induites par des processus stochastiques séparables à valeurs réelles et que $d$ est la norme supremum. En ce qui concerne le PGD, des mesures exponentiellement équivalentes sont indistinctes, comme l'atteste le théorème suivant.
	\end{itemize}
\end{rem}
Le théorème suivant, extrait de \cite{[8]}, affirme que deux familles de mesures exponentiellement équivalentes satisfont le même principe de grandes déviations.

\begin{thm}
	Si un principe de grandes déviations avec une bonne fonction de taux $I(\cdot)$ 
	est vérifié pour la famille de probabilités $\{\mu_\varepsilon\}$, et si 
	$\{\mu_\varepsilon\}$ est exponentiellement équivalente à $\{\tilde{\mu}_\varepsilon\}$, 
	alors le même principe de grandes déviations est vérifié pour $\{\tilde{\mu}_\varepsilon\}$.
	
\end{thm}
Le résultat suivant constitue une application du principe de contraction aux processus de diffusion. Il permet d’établir un principe de grandes déviations pour la solution d’une équation différentielle stochastique, à partir de celui vérifié par le mouvement brownien.

Soit $\{x_t^\varepsilon\}_{t \in [0,1]}$ l’unique solution de l’équation différentielle stochastique
\[
dx_t^\varepsilon = b(x_t^\varepsilon)\,dt + \sqrt{\varepsilon}\,dw_t, \quad x_0^\varepsilon = 0,
\]
où $b : \mathbb{R} \to \mathbb{R}$ est une fonction lipschitzienne.
